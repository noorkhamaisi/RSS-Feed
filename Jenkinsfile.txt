pipeline {

agent any

stages{

stage('Clone') {
 steps {
        
        git 'https://github.com/noorkhamaisi/RSS-Feed.git'
       
    }  
}
stage('Build') {
   steps {
    sh """
    
    . .env/bin/activate

    pip3 install -r requirements.txt
    """
    }  
}

stage('run') {
   steps {
     sh """ 
    . .env/bin/activate
    python3 main.py
    """

    }  
}

}

}
